class Publisher {
  constructor() {
    this.listeners = new Map();
  }

  subscribe(key, event) {
    if (this.listeners.has(key)) {
      this.listeners.set(key, this.listeners.get(key).push(event));
    } else {
      this.listeners.set(key, [event]);
    }
  }

  unSubscribe(key, event) {
    if (this.listeners.has(key)) {
      const events = this.listeners.get(key);
      const index = events.indexOf(event);
      if (index >= 0) {
        events.splice(index, 1);
        this.listeners.set(key, events);
      }
    }
  }

  notifyFunc(key, ...args) {
    if (this.listeners.has(key)) {
      this.listeners.get(key).map((listener) => listener.apply(null, args));
    }
  }

  notifyClass(key, ...args) {
    if (this.listeners.has(key)) {
      this.listeners.get(key).map((listener) => listener.update(...args));
    }
  }
}
